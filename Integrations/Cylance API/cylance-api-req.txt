#!/bin/bash

CYLANCE_TID="tenant-id"
CYLANCE_APP_ID="app-id"
CYLANCE_APP_SECRET="sec-key"
CYLANCE_REGION="-euc1" # ex. '-euc1'


TID="$CYLANCE_TID" # Cylance Tenant ID
APP_ID="$CYLANCE_APP_ID"
APP_SECRET="$CYLANCE_APP_SECRET"
REGION="$CYLANCE_REGION"

API_HOST="https://protectapi${REGION}.cylance.com"
AUTH_URL="${API_HOST}/auth/v2/token"
JTI_VAL=$(uuidgen)
TIMEOUT=1800
EPOCH_TIME=$(date +%s)
EPOCH_TIMEOUT=$(expr "$EPOCH_TIME" + "$TIMEOUT")

HEADER=$(echo -n '{"typ":"JWT","alg":"HS256"}' | /bin/base64 -w 0 | sed 's/+/-/g; s/\//_/g ; s/=\+$//g')

PAYLOAD=$(jq -n -c -r \
            --argjson exp "$EPOCH_TIMEOUT" \
            --argjson iat "$EPOCH_TIME" \
            --arg iss "http://cylance.com" \
            --arg sub "$APP_ID" \
            --arg tid "$TID" \
            --arg jti "$JTI_VAL" \
            '{"exp": $exp,"iat": $iat,"iss": $iss,"sub": $sub,"tid": $tid,"jti": $jti}' \
            | base64 -w 0 | sed 's/+/-/g; s/\//_/g; s/=\+$//g' )

SIGNATURE=$(echo -n "${HEADER}"."${PAYLOAD}" | openssl dgst -sha256 -hmac "${APP_SECRET}" -binary | openssl base64 -e -A | sed 's/+/-/g; s/\//_/g; s/=\+$//g')

TOKEN="${HEADER}.${PAYLOAD}.${SIGNATURE}"

CYLANCE_AUTH_TOKEN=$(jq -n -c \
                      --arg auth_token "$TOKEN" \
                      '{"auth_token": $auth_token}')

BEARER=$(curl -vv -s -X POST -H 'Content-Type: application/json; charset=utf-8' "$AUTH_URL" -d "$CYLANCE_AUTH_TOKEN" | jq -r .access_token)

# Example fetching some Device records from Cylance Protect API

DEVICES=$(curl -s --location --request GET "${API_HOST}/devices/v2?page=1&page_size=5000" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")

cdate=$(date +"%Y-%m-%d")T00:00:00.000Z

DETECTIONS=$(curl -s --location --request GET "${API_HOST}/detections/v2?page=1&page_size=100&start=${cdate}" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")


THREATS=$(curl -s --location --request GET "${API_HOST}/threats/v2?page=1_size=100" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")

MEMPRO=$(curl -s --location --request GET "${API_HOST}/memoryprotection/v2?page=1&page_size=200" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")

GLOBALLIST=$(curl -s --location --request GET "${API_HOST}/globallists/v2?listTypeId=0&page=1&page_size=100" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")

GLOBALLIST1=$(curl -s --location --request GET "${API_HOST}/globallists/v2?listTypeId=1&page=1&page_size=100" \
  --header 'Accept: application/json' \
  --header "Authorization: Bearer ${BEARER}")

#DETECTIONS1=$(curl -s --location --request GET "${API_HOST}/detections/v2?page=1&page_size=10&sort=OccurrenceTime" \
#  --header 'Accept: application/json' \
#  --header "Authorization: Bearer ${BEARER}")


#echo $DETECTIONS1 | jq -c -M '.page_items[]' > /etc/cylanceapi/cyl_detections.tmp

#echo $DEVICES | jq -c -M '.page_items[]' > /etc/cylanceapi/cyl_devices.txt


rm -f *.txt
rm -f *.tmp

echo $DETECTIONS | jq -c -M '.page_items[]' > /etc/cylanceapi/scyl_detections.tmp

echo $THREATS | jq -c -M '.page_items[]' > /etc/cylanceapi/scyl_threats.tmp

echo $MEMPRO | jq -c -M '.page_items[]' > /etc/cylanceapi/scyl_mempro.tmp

echo $GLOBALLIST | jq -c -M '.page_items[]' > /etc/cylanceapi/scyl_globallist.tmp

echo $GLOBALLIST1 | jq -c -M '.page_items[]' >> /etc/cylanceapi/scyl_globallist.tmp

sleep 1

cp -f /etc/cylanceapi/scyl_detections.tmp /etc/cylanceapi/cyl_detections.txt

cp -f /etc/cylanceapi/scyl_threats.tmp /etc/cylanceapi/cyl_threats.txt

cp -f /etc/cylanceapi/scyl_mempro.tmp /etc/cylanceapi/cyl_mempro.txt

cp -f /etc/cylanceapi/scyl_globallist.tmp /etc/cylanceapi/cyl_globallist.txt
